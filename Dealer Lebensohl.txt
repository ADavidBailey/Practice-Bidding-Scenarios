Script,Lebensohl
setDealerCode(
`
# Lebensohl

# Definitions
# Good Suits West
GSW = spades(west)>3 and top4(west,spades)>2
GHW = hearts(west)>3 and top4(west,hearts)>2
GDW = diamonds(west)>3 and top4(west,diamonds)>2
GCW = clubs(west)>3 and top4(west,clubs)>2

# Good Suits North
GSN = spades(north)>4 and top5(north,spades)>2
GHN = hearts(north)>4 and top5(north,hearts)>2
GDN = diamonds(north)>4 and top5(north,diamonds)>2
GCN = clubs(north)>4 and top5(north,clubs)>2

# South opens 1N
hcp(south)>14 and hcp(south)<18 and
shape(south,any 4333 +any 4432 +any 5332) and

# West competes
hcp(west)>10 and
(
# 2-suited
(shape(west,any 54xx +any 55xx) and
((GSW and GHW)
or (GSW and GDW)
or (GSW and GCW)
or (GHW and GDW)
or (GHW and GCW)
or (GDW and GCW))
)
or
# 1-suited
(shape(west,any 6xxx +any 5xxx) and
(GSW or GHW or GDW or GCW))
)
and

# North has something
hcp(north)>7 and
(shape(north, any 5xxx +any 4xxx) and
(GSN or GHN or GDN or GCN))

`, "S", true);
Script
Button,Lebensohl,\n\
Lebensohl\n\%Lebensohl%
