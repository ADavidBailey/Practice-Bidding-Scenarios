Script,Notrump
setDealerCode(
`
# 1 Notrump Opening

# Define Doubleton Honors
S2H = spades(south)==2 and (hascard(south,KS) + hascard(south,QS) + hascard(south,JS)) == 2
H2H = hearts(south)==2 and (hascard(south,KH) + hascard(south,QH) + hascard(south,JH)) == 2
D2H = diamonds(south)==2 and (hascard(south,KD) + hascard(south,QD) + hascard(south,JD)) == 2
C2H = clubs(south)==2 and (hascard(south,KC) + hascard(south,QC) + hascard(south,JC)) == 2

downgrade = (S2H or H2H or D2H or C2H) ? 1 : 0
adjustedTP = shape(south, any 5332) ? hcp(south) + 1 - downgrade : hcp(south)  // +1 unless we also have a doubleton honor

shape(south, any 4333+4432+5332) and adjustedTP>14 and adjustedTP<18

`
)
Script
Button,1 Notrump Opening,\n\
---  1 Notrump Opening\n\
4333，4432，or 5332 distribution\n\
HCP adjusted +1 for any 5-card suit unless you have KQ，KJ，or QJ doubleton.\n\
%Notrump%
