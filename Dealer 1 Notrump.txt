Script,Notrump
setDealerCode(
`
# 1 Notrump Opening

# Define good suits
gS = spades(south)==5 and (top5(south,spades)>2 or (top5(south,spades)==2 and hascard(south,9S)))
gH = hearts(south)==5 and (top5(south,hearts)>2 or (top5(south,hearts)==2 and hascard(south,9H)))
gD = diamonds(south)==5 and (top5(south,diamonds)>2 or (top5(south,diamonds)==2 and hascard(south,9D)))
gC = clubs(south)==5 and (top5(south,clubs)>2 or (top5(south,clubs)==2 and hascard(south,9C)))

sTP = (gS or gH or gD or gC) ? hcp(south) + 1 : hcp(south)    // +1 if we have a 'good' 5-card suit

shape(south, any 4333+4432+5332) and sTP>14 and sTP<18

`
)
Script
Button,1 Notrump Opening,\n\
---  1 Notrump Opening\n\
4333，4432，or 5332 distribution\n\
HCP adjusted for a good 5-card suit (>2 of the top 5 or =2 of the top 5 and the 9)\n\
%Notrump%
